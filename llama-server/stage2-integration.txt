# ── .env additions ────────────────────────────────────────────────────────────
# Run:  ls /data/models/gguf/Qwen3-Coder-Next-Q4_K_M/
# and set LLAMA_MODEL_FILE to whatever .gguf filename appears, e.g.:
#
#   LLAMA_MODEL_FILE=Qwen3-Coder-Next-Q4_K_M-00001-of-00004.gguf
#   # or for a single-shard model:
#   LLAMA_MODEL_FILE=Qwen3-Coder-Next-Q4_K_M.gguf
#
# MoE expert offload (three expert weight tensor families → system RAM).
# Leave blank to keep everything on GPU (will OOM for large ctx on 48 GB).
LLAMA_OVERRIDE_TENSOR=ffn_down_exps=CPU ffn_gate_exps=CPU ffn_up_exps=CPU

# Context window. 32 k fits comfortably alongside the expert offload on 48 GB.
# Bump to 65536 cautiously — watch nvidia-smi for VRAM headroom.
LLAMA_CTX_SIZE=32768

# Flash attention — recommended for Qwen3 / GQA architectures
LLAMA_FLASH_ATTN=1

# Threads for the CPU-side expert compute (MoE offload runs on CPU).
# Set to physical core count on your machine.
LLAMA_THREADS=16


# ── docker-compose.yml service block ─────────────────────────────────────────
# Drop this into the `services:` section of your existing docker-compose.yml.
# It replaces the stub llama-server service.

  llama-server:
    build:
      context: ./llama-server        # directory containing the Dockerfile
      dockerfile: Dockerfile
      args:
        LLAMA_TAG: b9204             # pin to verified release
        CUDA_ARCH: "86"              # RTX A6000 = sm_86
        BUILD_JOBS: "0"              # 0 → nproc inside the builder stage
    image: llm-stack/llama-server:b9204
    restart: unless-stopped
    ports:
      - "127.0.0.1:8000:8000"        # LAN access comes via LiteLLM proxy only
    volumes:
      - ${LLAMA_MODEL_DIR}:/models:ro  # bind-mount host model dir read-only
    environment:
      LLAMA_MODEL_FILE:    ${LLAMA_MODEL_FILE}
      LLAMA_OVERRIDE_TENSOR: ${LLAMA_OVERRIDE_TENSOR}
      LLAMA_CTX_SIZE:      ${LLAMA_CTX_SIZE:-32768}
      LLAMA_FLASH_ATTN:    ${LLAMA_FLASH_ATTN:-1}
      LLAMA_THREADS:       ${LLAMA_THREADS:-8}
      LLAMA_N_GPU_LAYERS:  ${LLAMA_N_GPU_LAYERS:-999}
    deploy:
      resources:
        reservations:
          devices:
            - driver: nvidia
              count: 1
              capabilities: [gpu]
    healthcheck:
      test: ["CMD", "curl", "-sf", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      start_period: 120s    # large model takes time to load
      retries: 3
